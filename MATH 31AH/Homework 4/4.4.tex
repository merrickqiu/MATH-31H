\section{Representing matrices}
\begin{align*}
    \frac{d}{dx}(c_1x^3 + c_2x^2 + c_3x + c_4) 
    &= 3c_1x^2 + 2c_2x + c_3 \\ 
    &= (3c_1)(x+1)^2 + (-6c_1+2c_2)(x+1) + (3c_1-2c_2+c_3)(1)
\end{align*}

Therefore,

\[
[T]_\mathcal{C}^\mathcal{B} =
\begin{bmatrix}
    3 & 0 & 0 & 0\\
    -6 & 2 & 0 & 0\\
    3 & -2 & 1 & 0\\
\end{bmatrix} 
\].
